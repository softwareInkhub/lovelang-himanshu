A live preview of the section

An editable view of the code (HTML/JSX, CSS, and optional JS)

A side panel to update section configs (like text, images, background)

API support to fetch/upload code for each section (Node.js or AWS Lambda backend)

Create a builder page like /pages/builder.jsx:

jsx
Copy
Edit
const Builder = () => {
  const [selectedSection, setSelectedSection] = useState('HeroSection');
  const [config, setConfig] = useState({});
  const [code, setCode] = useState('');

  useEffect(() => {
    fetch(`/api/components/${selectedSection}`).then(res => res.json()).then(data => {
      setConfig(data.config);
      setCode(data.code);
    });
  }, [selectedSection]);

  const saveChanges = () => {
    fetch(`/api/components/${selectedSection}`, {
      method: 'POST',
      body: JSON.stringify({ code, config }),
      headers: { 'Content-Type': 'application/json' }
    });
  };

  return (
    <div className="grid grid-cols-2 h-screen">
      <div className="overflow-auto p-4">
        <ComponentPreview code={code} config={config} />
      </div>
      <div className="overflow-auto border-l p-4 space-y-4">
        <h2 className="text-xl font-semibold">{selectedSection} Code</h2>
        <CodeEditor value={code} onChange={setCode} />
        <ConfigForm config={config} onChange={setConfig} />
        <button onClick={saveChanges}>Save</button>
      </div>
    </div>
  );
};

Copy

const fs = require('fs');
const path = require('path');

exports.getComponent = (req, res) => {
  const name = req.params.name;
  const jsxPath = path.join(__dirname, `../../components/${name}.jsx`);
  const configPath = path.join(__dirname, `../../configs/${name.toLowerCase()}.json`);

  const code = fs.readFileSync(jsxPath, 'utf-8');
  const config = JSON.parse(fs.readFileSync(configPath, 'utf-8'));

  res.json({ code, config });
};

exports.updateComponent = (req, res) => {
  const name = req.params.name;
  const jsxPath = path.join(__dirname, `../../components/${name}.jsx`);
  const configPath = path.join(__dirname, `../../configs/${name.toLowerCase()}.json`);

  fs.writeFileSync(jsxPath, req.body.code);
  fs.writeFileSync(configPath, JSON.stringify(req.body.config, null, 2));
  res.json({ success: true });
};
